172 double CPlant::ReproGrow(double uptake) {
173  double SpacerRes, SeedRes, VegRes, dm_seeds, dummy1;
174  //fixed Proportion of resource to seed production
175  if (mRepro <= Traits->AllocSeed * mshoot) //calculate mRepro for every week
176  {
177  SeedRes = uptake * Traits->AllocSeed;
178  SpacerRes = uptake * Traits->AllocSpacer;
179 
180  int pweek = CEnvir::week;
181 
182  //during the seed-production-weeks
183  if ((pweek >= Traits->FlowerWeek) && (pweek < Traits->DispWeek)) {
184  //seed production
185  dm_seeds = max(0.0, Traits->growth * SeedRes);
186  mRepro += dm_seeds;
187 
188  //clonal growth
189  dummy1 = max(0.0, min(SpacerRes, uptake - SeedRes)); // for large AllocSeed, ressources may be < SpacerRes, then only take remaining ressources
190  mReproRamets += max(0.0, Traits->growth * dummy1);
191  VegRes = uptake - SeedRes - dummy1;
192 
193  } else {
194  VegRes = uptake - SpacerRes;
195  mReproRamets += max(0.0, Traits->growth * SpacerRes);
196  }
197 
198  } else
199  VegRes = uptake;
200  return VegRes;
201 } //end reprogrow

206 void CPlant::SpacerGrow()
207 {
208  double mGrowSpacer=0;
209  int SpacerListSize=this->growingSpacerList.size();
210 
211  if (SpacerListSize==0)return;
212  if ((mReproRamets>0))
213  {
214  mGrowSpacer=(mReproRamets/SpacerListSize);//resources for one spacer
215 
216  for (int g=0; g<(SpacerListSize); g++)
217  { //loop for all growing Spacer of one plant
218  CPlant* Spacer = this->growingSpacerList[g];
219 
220  double lengthtogrow=Spacer->SpacerlengthToGrow;
221  lengthtogrow-=(mGrowSpacer/Traits->mSpacer); //spacer growth
222  Spacer->SpacerlengthToGrow=max(0.0,lengthtogrow);
223 
224  //Estab for all growing Spacers in the last week of the year
225  if ((CEnvir::week==CEnvir::WeeksPerYear)
226  && (Spacer->SpacerlengthToGrow>0))
227  {
228  double direction=Spacer->Spacerdirection;
229  double complDist=Spacer->Spacerlength;//should be positive
230  double dist=(complDist-Spacer->SpacerlengthToGrow);
231  double CmToCell=1.0/SRunPara::RunPara.CellScale();
232  int x2=CEnvir::Round(this->cell->x+cos(direction)*dist*CmToCell);
233  int y2=CEnvir::Round(this->cell->y+sin(direction)*dist*CmToCell);
234 
236  //Boundary conditions (types MUST match exactly)
237  Boundary(x2,y2);
238 // CGrid::Boundary(x2,y2);
239 
240  //beachte dass CellNum nicht immer == CellSize
241  Spacer->xcoord=x2/CmToCell;
242  Spacer->ycoord=y2/CmToCell;
243  Spacer->SpacerlengthToGrow=0;
244  } //end if pweek==WeeksPerYear
245  } //end List of Spacers
246  }
247  mReproRamets=0;
248 } //end SpacerGrow

262 void CPlant::Grow2() //grow plant one timestep
263 {
264  double dm_shoot, dm_root,alloc_shoot;
265 // double Assim_shoot, Resp_shoot, Assim_root, Resp_root;
266  double Assim_shoot,Assim_root,Resp;
267  double LimRes, ShootRes, RootRes, VegRes;
268  double p=2.0/3.0, q=2.0, r=4.0/3.0; //exponents for growth function
269  int pweek=CEnvir::week;
270 
271  /********************************************/
272  /* dm/dt = growth*(c*m^p - m^q / m_max^r) */
273  /********************************************/
274 
275  //which resource is limiting growth ?
276  LimRes=min(Buptake,Auptake); //two layers
277  VegRes=ReproGrow(LimRes);
278 
279  //allocation to shoot and root growth
280  alloc_shoot= Buptake/(Buptake+Auptake); //allocation coefficient
281 
282  ShootRes=alloc_shoot*VegRes;
283  RootRes =VegRes-ShootRes;
284 
285  //Shoot growth
286  dm_shoot=this->ShootGrow(ShootRes);
287 
288  //Root growth
289  dm_root=this->RootGrow(RootRes);
290 
291  mshoot+=dm_shoot;
292  mroot+=dm_root;
293 
294  if (stressed())++stress;
295  else if (stress>0) --stress;
296 }
302 double CPlant::ShootGrow(double shres){
303  double Assim_shoot, Resp_shoot;
304  double p=2.0/3.0, q=2.0, r=4.0/3.0; //exponents for growth function
305  Assim_shoot=Traits->growth*min(shres,Traits->Gmax*Ash_disc); //growth limited by maximal resource per area -> similar to uptake limitation
306  Resp_shoot=Traits->growth*Traits->SLA
307  *pow(Traits->LMR,p)*Traits->Gmax
308  *pow(mshoot,q)/pow(Traits->MaxMass,r); //respiration proportional to mshoot^2
309  return max(0.0,Assim_shoot-Resp_shoot);
310 }
316 double CPlant::RootGrow(double rres){
317  double Assim_root, Resp_root;
318  double p=2.0/3.0, q=2.0, r=4.0/3.0; //exponents for growth function
319  Assim_root=Traits->growth*min(rres,Traits->Gmax*Art_disc); //growth limited by maximal resource per area -> similar to uptake limitation
320  Resp_root=Traits->growth*Traits->Gmax*Traits->RAR
321  *pow(mroot,q)/pow(Traits->MaxMass,r); //respiration proportional to root^2
322 
323  return max(0.0,Assim_root-Resp_root);
324 }
